\ProblemTitle{Find Divisors (P9) â€” All Divisors of an Integer}
\Difficulty{Easy}
\Tags{easy, math, number theory}
\Authors{Gemini}

\Statement{
You are given a positive integer $n$.
Your task is to print all the **divisors** of $n$ in **ascending order**.
}

\InputFormat{A single integer $n$.}
\OutputFormat{Print all divisors of $n$ in ascending order, separated by spaces.}
\Constraints{$1 \leq n \leq 10^9$.}

\noindent\textbf{Sample Input 1:}
\\begin{verbatim}
12
\end{verbatim}
\\noindent\textbf{Sample Output 1:}
\\begin{verbatim}
1 2 3 4 6 12
\end{verbatim}

\\noindent\textbf{Sample Input 2:}
\\begin{verbatim}
7
\end{verbatim}
\\noindent\textbf{Sample Output 2:}
\\begin{verbatim}
1 7
\end{verbatim}

\\noindent\textbf{Sample Input 3:}
\\begin{verbatim}
1
\end{verbatim}
\\noindent\textbf{Sample Output 3:}
\\begin{verbatim}
1
\end{verbatim}

\\noindent\textbf{Notes:} The output should be a single line of space-separated integers.
