\ProblemTitle{Odd or Even (P6)}
\Difficulty{Easy}
\Tags{easy, math, implementation}
\Authors{Admin}

\Statement{
Given an integer, you need to determine whether it is ODD or EVEN.
You are provided with multiple test cases, and for each test case, you must print "ODD" if the number is odd and "EVEN" if the number is even.
}

\InputFormat{The first line contains a single integer $t$ â€” the number of test cases.
Each of the next $t$ lines contains a single integer $n$.}
\OutputFormat{For each test case, print "ODD" if $n$ is odd, otherwise print "EVEN".}
\Constraints{$1 \leq t \leq 10^5$, $0 \leq n \leq 10^{18}$.}

\noindent\\textbf{Sample Input 1:}
\\begin{verbatim}

2
3
10
15
\\end{verbatim}
\\noindent\\textbf{Sample Output 1:}
\\begin{verbatim}

ODD
EVEN
ODD
\\end{verbatim}
