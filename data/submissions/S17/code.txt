const int INF = (int) 1e9;

const ll MOD = 998244353;


struct DSU {
    vector<int> p, r;
    int n;
    DSU(int n_nodes = 0) : n(n_nodes), p(n_nodes + 1), r(n_nodes + 1, 0) {
        for (int i = 0; i <= n; ++i) {
            p[i] = i;
        }
    }
    int f(int x) { return p[x] == x ? x : p[x] = f(p[x]); }
    void un(int a, int b) {
        a = f(a);
        b = f(b);
        if (a == b) return;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
    }
};

long long power(long long base, long long exp) {
    long long ans = 1;
    base %= MOD;

    while (exp) {
        if (exp & 1) ans = (ans * base) % MOD;
        base = (base * base) % MOD;
        exp >>= 1;
    }

    return ans;
}
